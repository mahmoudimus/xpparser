\begin{figure*}[htp]
  \centering
  % you need to compile with `pdflatex --shell-escape`
  \begin{gnuplot}[terminal=cairolatex, terminaloptions={color size
        11.0cm,6.0cm}]
    set decimal locale
    set format y "%'g"
    set yrange [0 to 15000]
    set style fill solid 1.00 
    set ylabel "number of queries"
    set xtics nomirror rotate by -45
    set tic scale 0
    set grid ytics lc rgb "#505050"
    unset border
    set style data histograms
    set boxwidth 1.3
    set key font ",9"
    plot "xslt.dat" using 2:xticlabels(1) notitle lt rgb "#202020",\
    "" using 3 title "+extras" lt rgb "#fdc711", "" using 4 title "extended" lt rgb "#5f1d5e", "" using\
    5 title "baseline" lt rgb "#636369"
  \end{gnuplot}
  \caption{Coverage of the XSLT sources.}
\end{figure*}
